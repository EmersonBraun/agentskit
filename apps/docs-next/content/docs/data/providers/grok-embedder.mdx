```ts
import { grokEmbedder } from '@agentskit/adapters'

const embed = grokEmbedder({

})
```

## Options

| Option | Type | Default |
|---|---|---|
| `apiKey` | `string` | required |
| `model` | `string` | provider default |
| `baseUrl` | `string` | `https://api.x.ai/v1` |
